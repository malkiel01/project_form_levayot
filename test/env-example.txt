# .env.example - דוגמה לקובץ הגדרות סביבה
# יש להעתיק קובץ זה ל-.env ולעדכן את הערכים

# הגדרות דטהבייס
DB_HOST=localhost
DB_NAME=cemetery_db
DB_USER=your_database_user
DB_PASS=your_database_password
DB_CHARSET=utf8mb4

# הגדרות אתר
SITE_URL=https://yourdomain.com
SITE_NAME="מערכת ניהול נפטרים"
SITE_EMAIL=info@yourdomain.com
UPLOAD_PATH=/var/www/html/uploads/
TIMEZONE=Asia/Jerusalem

# הגדרות אבטחה
ENCRYPTION_KEY=your-32-character-encryption-key-here
API_SECRET=your-api-secret-key-here
SESSION_LIFETIME=3600
CSRF_TOKEN_LIFETIME=3600

# הגדרות דואר אלקטרוני
MAIL_DRIVER=smtp
MAIL_HOST=smtp.gmail.com
MAIL_PORT=587
MAIL_USERNAME=your-email@gmail.com
MAIL_PASSWORD=your-app-password
MAIL_ENCRYPTION=tls
MAIL_FROM_ADDRESS=noreply@yourdomain.com
MAIL_FROM_NAME="${SITE_NAME}"

# הגדרות SMS (אופציונלי)
SMS_PROVIDER=twilio
SMS_ACCOUNT_SID=your-twilio-account-sid
SMS_AUTH_TOKEN=your-twilio-auth-token
SMS_FROM_NUMBER=+1234567890

# הגדרות גיבוי
BACKUP_ENABLED=true
BACKUP_PATH=/var/backups/cemetery/
BACKUP_RETENTION_DAYS=30
BACKUP_EMAIL_NOTIFICATION=admin@yourdomain.com

# הגדרות לוג
LOG_LEVEL=info
LOG_PATH=/var/log/cemetery/
LOG_MAX_FILES=30

# הגדרות ביצועים
CACHE_ENABLED=true
CACHE_DRIVER=file
CACHE_PATH=/var/cache/cemetery/
CACHE_TTL=3600

# הגדרות פיתוח/ייצור
APP_ENV=production
APP_DEBUG=false
DISPLAY_ERRORS=false

# הגדרות API
API_RATE_LIMIT=100
API_RATE_LIMIT_WINDOW=3600

# הגדרות תמונות
IMAGE_MAX_WIDTH=2000
IMAGE_MAX_HEIGHT=2000
IMAGE_QUALITY=85
IMAGE_ALLOWED_TYPES=jpg,jpeg,png,gif

# הגדרות PDF
PDF_FONT=dejavusans
PDF_FONT_SIZE=11
PDF_MARGIN_TOP=25
PDF_MARGIN_BOTTOM=25
PDF_MARGIN_LEFT=15
PDF_MARGIN_RIGHT=15

# הגדרות מערכת
MAINTENANCE_MODE=false
MAINTENANCE_MESSAGE="המערכת בתחזוקה, נשוב בקרוב"
DEFAULT_LANGUAGE=he
SUPPORTED_LANGUAGES=he,en,ar,ru
